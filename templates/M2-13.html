1หลักการออกแบบ (Design Principles)

  • Normalization (3NF) — ออกแบบตารางให้อยู่ในรูปแบบบรรทัดฐานที่สาม ลดความซ้ำซ้อน (redundancy) และความผิดปกติของข้อมูล (anomaly) ยกเว้นกรณีจำเป็นด้านประสิทธิภาพจึง denormalize อย่างมีเหตุผล
  • Referential Integrity — บังคับความสัมพันธ์ด้วย Foreign Key Constraint ทุกความสัมพันธ์ พร้อมกำหนด ON DELETE/UPDATE ที่เหมาะสม (RESTRICT/CASCADE)
  • Surrogate Key — ทุกตารางใช้ <table>_id (BIGINT) เป็น PK เพื่อความเสถียร ส่วน Business Key (เช่น HN, VN) ใช้ UNIQUE Constraint
  • Data Type ข้ามแพลตฟอร์ม — ใช้ชนิดข้อมูลมาตรฐาน ISO SQL รองรับ PostgreSQL / SQL Server / Oracle (ค่าเงิน NUMERIC, เวลา TIMESTAMP)
  • Audit & Soft Delete — ทุกตารางหลักมี created_at, updated_at และพิจารณา is_deleted สำหรับ soft delete ข้อมูลผู้ป่วย
  • ความปลอดภัยข้อมูล — ข้อมูล PII/สุขภาพ เข้ารหัสและควบคุมการเข้าถึงตามสิทธิ์ (สอดคล้อง PDPA)

2แผนภาพความสัมพันธ์ของข้อมูล (ERD)

แสดงความสัมพันธ์หลักของเอนทิตี โดยใช้สัญลักษณ์ 1 ──< N หมายถึงความสัมพันธ์แบบหนึ่งต่อกลุ่ม (One-to-Many)

หมายเหตุ: patient 1:N visit, visit 1:N treatment, patient 1:N appointment, visit 1:N prescription, prescription 1:N prescription_item (เชื่อม drug), users (dentist/student) 1:N visit/treatment

4นิยามตารางตัวอย่าง (Table Definition — DDL)

ตัวอย่าง DDL เชิงตรรกะ (ปรับชนิดข้อมูลให้ตรง DBMS ที่ใช้จริง) — แสดง field, type, PK/FK

CREATE TABLE patient ( patient_id BIGINT NOT NULL, hn VARCHAR(13) NOT NULL, citizen_id VARCHAR(13) NULL, title VARCHAR(20) NULL, first_name VARCHAR(100) NOT NULL, last_name VARCHAR(100) NOT NULL, gender CHAR(1) NOT NULL, -- M / F / U birth_date DATE NULL, phone VARCHAR(20) NULL, allergy_flag BOOLEAN NOT NULL DEFAULT FALSE, created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP, updated_at TIMESTAMP NULL, CONSTRAINT pk_patient PRIMARY KEY (patient_id), CONSTRAINT uq_patient_hn UNIQUE (hn) ); CREATE TABLE visit ( visit_id BIGINT NOT NULL, patient_id BIGINT NOT NULL, visit_no VARCHAR(15) NOT NULL, visit_date TIMESTAMP NOT NULL, clinic_code VARCHAR(10) NOT NULL, dentist_id BIGINT NULL, visit_type VARCHAR(15) NOT NULL, status VARCHAR(15) NOT NULL, created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP, CONSTRAINT pk_visit PRIMARY KEY (visit_id), CONSTRAINT uq_visit_no UNIQUE (visit_no), CONSTRAINT fk_visit_patient FOREIGN KEY (patient_id) REFERENCES patient (patient_id), CONSTRAINT fk_visit_dentist FOREIGN KEY (dentist_id) REFERENCES users (user_id) ); CREATE TABLE treatment ( treatment_id BIGINT NOT NULL, visit_id BIGINT NOT NULL, tooth_no VARCHAR(3) NULL, -- FDI เช่น 11, 36, 48 surface VARCHAR(5) NULL, -- M/O/D/B/L procedure_code VARCHAR(15) NOT NULL, dx_code VARCHAR(15) NULL, -- ICD-10 เช่น K02.1 fee NUMERIC(12,2) NOT NULL DEFAULT 0.00, status VARCHAR(15) NOT NULL, performed_by BIGINT NULL, created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP, CONSTRAINT pk_treatment PRIMARY KEY (treatment_id), CONSTRAINT fk_treatment_visit FOREIGN KEY (visit_id) REFERENCES visit (visit_id) ON DELETE CASCADE, CONSTRAINT fk_treatment_proc FOREIGN KEY (procedure_code) REFERENCES procedure (code), CONSTRAINT fk_treatment_performer FOREIGN KEY (performed_by) REFERENCES users (user_id) );
